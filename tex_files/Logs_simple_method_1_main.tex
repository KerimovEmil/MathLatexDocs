\subsection{Exponating rules}\label{Exponating rules}
\begin{equation}\label{a_x_times_a_b}
a^{x} \cdot a^{y}  = a^{x+y}
\end{equation}
\begin{equation}\label{a_0}
a^{x} \cdot a^{-x} = a^{x} \cdot \frac{1}{a^{x}} = 1 = a^{x-x} = a^{0}
\end{equation}
